def extract_suspect_info(narrative):
    """Extract suspect description from narrative"""
    if pd.isna(narrative) or not narrative:
        return "Data Incomplete"
    
    # Look for "Suspect description:" pattern
    match = re.search(r'[Ss]uspect [Dd]escription:?\s*(.+?)(? :\n\n|\n     |$)', narrative, re.DOTALL)
    if match:
        desc = match.group(1).strip()
        # Clean up excessive whitespace
        desc = re.sub(r'\s+', ' ', desc)
        return desc if desc else "Data Incomplete"
    
    return "Data Incomplete"

def extract_loss_items(narrative, loss_total):
    """Extract loss items from narrative"""
    if pd.isna(narrative) or not narrative:
        return []
    
    items = []
    
    # Look for patterns like "concealed several iPhones and Macbooks"
    # This is basic extraction - AI would do better
    item_patterns = [
        r'(?:conceal(?:ed)?|stole|stolen|took)\s+(?:several\s+)? ([^.,\n]+)',
        r'(?:iphone|macbook|laptop|phone|tablet|cash|jewelry|wallet)s?' ]
    
    for pattern in item_patterns:
        matches = re.finditer(pattern, narrative, re.IGNORECASE)
        for match in matches:
            if match.group(0):
                item_text = match.group(0).strip()
                if len(item_text) > 3 and len(item_text) < 100:  # Reasonable length
                    items.append(item_text)
    
    # If we found items but no values, create generic entry
    if items and loss_total and loss_total != "$0.00":
        return [f"{', '.join(set(items[:3]))} - {loss_total}"]
    elif not items and loss_total and loss_total != "$0.00":
        return [f"Property - {loss_total}"]
    
    return []

def generate_case_card_html(row, category):
    """Generate HTML for a single case card"""
    # Extract data with safe string conversion
    case_num = str(row.get('CaseNumber', 'Data Incomplete'))
    incident_datetime = str(row.get('IncidentDateTime', 'Data Incomplete'))
    incident_type = str(row.get('IncidentType', 'Data Incomplete'))
    subtype = str(row.get('Subtype', '')) if pd.notna(row.get('Subtype')) else ''
    statute = str(row.get('Statute', '')) if pd.notna(row.get('Statute')) else ''
    full_address = str(row.get('FullAddress', 'Data Incomplete'))
    grid = str(row.get('Grid', '')) if pd.notna(row.get('Grid')) else ''
    # Format Zone as whole number (remove decimal if present)
    zone_raw = row.get('Zone', '')
    if pd.notna(zone_raw) and zone_raw != '':
        try:
            # Convert to float first to handle string numbers, then to int to remove decimals
            zone = str(int(float(zone_raw)))
        except (ValueError, TypeError):
            zone = str(zone_raw)
    else:
        zone = ''
    narrative = str(row.get('Narrative', '')) if pd.notna(row.get('Narrative')) else ''
    status = str(row.get('Status', 'Data Incomplete'))
    loss_total = str(row.get('LossTotal', '$0.00'))
    
    # Vehicle info with safe conversion
    victim_vehicle_make = str(row.get('VictimVehicleMake', '')) if pd.notna(row.get('VictimVehicleMake')) else ''
    victim_vehicle_model = str(row.get('VictimVehicleModel', '')) if pd.notna(row.get('VictimVehicleModel')) else ''
    plate = str(row.get('Plate', '')) if pd.notna(row.get('Plate')) else ''
    plate_state = str(row.get('PlateState', '')) if pd.notna(row.get('PlateState')) else ''
    
    # Build header line
    display_type = incident_type
    if subtype and subtype.strip():
        display_type += f" / {subtype}"
    if statute and statute.strip():
        display_type += f" - {statute}"
    
    header = f"Case {case_num} | {incident_datetime} | {display_type}"
    
    # Build bullet points based on category
    bullets = []
    
    # Location (always include)
    if full_address and full_address != 'Data Incomplete':
        bullets.append(f"<span class='lb'>Location:</span> {full_address}")
    
    # Grid/Zone
    if grid or zone:
        grid_zone = f"{grid if grid else 'N/A'} / {zone if zone else 'N/A'}"
        bullets.append(f"<span class='lb'>Grid/Zone:</span> {grid_zone}")
    
    # Category-specific bullets
    if category[1] in ['MOTOR VEHICLE THEFT', 'BURGLARY - AUTO']:
        # Vehicle info
        if victim_vehicle_make or victim_vehicle_model:
